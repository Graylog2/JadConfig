15  public class StringListConverter implements Converter<List<String>> {

19      /**
20       * Returns a {@link List} of {@link String}s representing the specified {@link String} {@literal value}.
21       * <p/>
22       * The {@literal value} is being split on comma.
23       *
24       * @param value The configuration parameter's {@link String} value
25       * @return A {@link List} of {@link String}s representing the configuration parameter's value
26       */
27      @Override
28      public List<String> convertFrom(String value) {
29  
30          if (value == null) {
31              throw new ParameterException("Couldn't convert value \"" + value + "\" to list of String.");
32          }
33  
34          if ("".equals(value)) {
35  
36              return Collections.emptyList();
37          }
38  
39          return Arrays.asList(value.split(DELIMITER));
40      }
41  
42      /**
43       * Returns a {@link String} instance representing the configuration parameter's {@literal value}.
44       * <p/>
45       * The elements of the provided {@link List} of {@link String}s are concatenated with commas.
46       *
47       * @param value The configuration parameter's {@link List} of {@link String} representation
48       * @return A {@link String} instance representing the configuration parameter's typed value
49       */
50      @Override
51      public String convertTo(List<String> value) {
52  
53          if (value == null) {
54              throw new ParameterException("Couldn't convert \"null\" to String.");
55          }
56  
57          StringBuilder sb = new StringBuilder();
58          String separator = "";
59  
60          for (String element : value) {
61              sb.append(separator);
62              sb.append(element);
63              separator = DELIMITER;
64          }
65  
66          return sb.toString();
67      }
