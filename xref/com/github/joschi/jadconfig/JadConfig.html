15  /**
16   * The main class for JadConfig. It's responsible for parsing the configuration bean(s) that contain(s) the annotated
17   * fields, use a {@link Repository} to read the raw configuration and assign the fields with the correct values.
18   * <p/>
19   * The configuration bean(s) you pass in the constructor are expected to have one or more {@link Parameter} annotations
20   * on them.
21   * <p/>
22   * You can pass either a single configuration bean or an array of objects. In the case of an array JadConfig will
23   * collect the {@link Parameter} annotations from all the objects passed in.
24   *
25   * @author jschalanda
26   */

65      /**
66       * Processes the configuration provided by the configured {@link Repository} and filling the provided configuration
67       * beans.
68       *
69       * @throws RepositoryException If an error occurred while reading from the configured {@link Repository}
70       * @throws ValidationException If any parameter couldn't be successfully validated
71       */
72      public void process() throws RepositoryException, ValidationException {
73  
74          LOG.info("Opening repository {}", repository);
75          repository.open();
76  
77          for (Object configurationBean : configurationBeans) {
78              LOG.info("Processing configuration bean {}", configurationBean);
79  
80              processClassFields(configurationBean, configurationBean.getClass().getDeclaredFields());
81              invokeValidatorMethods(configurationBean, configurationBean.getClass().getMethods());
82          }
83      }

85      private void processClassFields(Object configurationBean, Field[] fields) throws ValidationException {
86          for (Field field : fields) {
87              Parameter parameter = field.getAnnotation(Parameter.class);
88  
89              if (parameter != null) {
90                  LOG.debug("Processing field {}", parameter);
91  
92                  Object fieldValue = getFieldValue(field, configurationBean);
93  
94                  String parameterName = parameter.value();
95                  String parameterValue = repository.read(parameterName);
96  
97                  if (parameterValue == null && fieldValue == null && parameter.required()) {
98                      LOG.warn("Required parameter {} not found", parameterName);
99                      throw new ParameterException("Required parameter \"" + parameterName + "\" not found.");
100                 }
101 
102                 if (parameterValue != null) {
103 
104                     if (parameter.trim()) {
105                         LOG.debug("Trimmed parameter value {}", parameterName);
106                         parameterValue = Strings.trim(parameterValue);
107                     }
108 
109                     LOG.debug("Validating parameter {}", parameterName);
110                     validateParameter(parameter.validator(), parameterName, parameterValue);
111 
112                     LOG.debug("Converting parameter value {}: ", parameterName, parameterValue);
113                     fieldValue = convertStringValue(field.getType(), parameter.converter(), parameterValue);
114                 }
115 
116                 LOG.debug("Setting parameter {} to {}", parameterName, parameterValue);
117 
118                 try {
119                     field.set(configurationBean, fieldValue);
120                 } catch (Exception e) {
121                     throw new ParameterException("Couldn't set field " + field.getName(), e);
122                 }
123             }
124         }
125     }

128         Converter converter = getConverter(fieldType, converterClass);

130         LOG.debug("Loaded converter class for type {}: {}", fieldType, converter);

135     private Object getFieldValue(Field field, Object bean) {
136         field.setAccessible(true);
137 
138         try {
139             return field.get(bean);
140         } catch (IllegalAccessException e) {
141             throw new ParameterException("Couldn't obtain value of field " + field.getName(), e);
142         }
143     }

145     private Converter getConverter(Class<?> fieldType, Class<? extends Converter<?>> converterClass) {
146 
147         LOG.debug("Trying to find converter class {} for type {}", converterClass, fieldType);
148 
149         Class<? extends Converter<?>> clazz = converterClass;
150 
151         if (clazz == null || clazz == NoConverter.class) {
152             clazz = findConverter(fieldType);
153         }
154 
155         // Fallback to StringConverter
156         if (clazz == null) {
157             clazz = StringConverter.class;
158 
159             LOG.debug("Using fallback converter: {}", clazz);
160         }
161 
162         try {
163             return clazz.newInstance();
164         } catch (Exception e) {
165             throw new ParameterException("Couldn't initialize converter class " + clazz.getCanonicalName(), e);
166         }
167     }

169     private void validateParameter(Class<? extends Validator> validatorClass, String name, String value) throws ValidationException {
170         Validator validator;
171 
172         LOG.debug("Validating parameter {} with value {}", name, value);
173 
174         try {
175             validator = validatorClass.newInstance();
176         } catch (Exception e) {
177             throw new ParameterException("Couldn't initialize validator " + validatorClass.getCanonicalName(), e);
178         }
179 
180         validator.validate(name, value);
181     }

183     private void invokeValidatorMethods(Object configurationBean, Method[] methods) throws ValidationException {
184         for (Method method : methods) {
185             if (method.isAnnotationPresent(ValidatorMethod.class)) {
186                 LOG.debug("Invoking validator method {} in {}", method, configurationBean);
187 
188                 try {
189                     method.invoke(configurationBean);
190                 } catch (Exception e) {
191                     throw new ValidationException("Couldn't run validator method " + method.getName(), e);
192                 }
193             }
194         }
195     }

197     private <T> Class<? extends Converter<T>> findConverter(Class<T> clazz) {
198         for (ConverterFactory factory : converterFactories) {
199             Class<? extends Converter<T>> result = factory.getConverter(clazz);
200 
201             if (result != null) {
202                 return result;
203             }
204         }
205 
206         return null;
207     }

231     /**
232      * Saves all configuration parameters to the configured {@link Repository}.
233      *
234      * @throws RepositoryException If an error occurred while writing to or saving the configured {@link Repository}
235      */
236     public void save() throws RepositoryException {
237         LOG.info("Saving changed configuration parameters");
238 
239         for (Object configurationBean : configurationBeans) {
240             LOG.debug("Checking declared fields of {}", configurationBean);
241 
242             for (Field field : configurationBean.getClass().getDeclaredFields()) {
243                 Parameter parameter = field.getAnnotation(Parameter.class);
244 
245                 LOG.debug("Checking declared field {} of {}", parameter, configurationBean);
246 
247                 if (parameter != null) {
248 
249                     Object fieldValue = getFieldValue(field, configurationBean);
250 
251                     LOG.debug("Retrieved field value {} = {}", field, fieldValue);
252 
253                     if (fieldValue != null) {
254                         String stringValue = convertFieldValue(field.getType(), parameter.converter(), fieldValue);
255 
256                         LOG.debug("Writing {} = {} to repository", parameter.value(), stringValue);
257                         repository.write(parameter.value(), stringValue);
258                     }
259                 }
260             }
261         }
262 
263         LOG.debug("Saving changes to repository {}", repository);
264         repository.save();
265     }

267     private String convertFieldValue(Class<?> fieldType, Class<? extends Converter<?>> converterClass, Object fieldValue) {
268         Converter converter = getConverter(fieldType, converterClass);
269 
270         LOG.debug("Converting {} to type {} using converter {}", new Object[]{fieldValue, fieldType, converter});
271         return converter.convertTo(fieldValue);
272     }
